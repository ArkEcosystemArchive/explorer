import Vue from "vue";
import VueI18n, { LocaleMessages } from "vue-i18n";
 
Vue.use(VueI18n);
 
function loadLocaleMessages(): LocaleMessages {
  const locales = require.context("./locales", true, /[A-Za-z0-9-_,\s]+\.ts$/i);
  const messages: LocaleMessages = {};
  locales.keys().forEach(key => {
    const matched = key.match(/([A-Za-z0-9-_]+)\./i);
    if (matched && matched.length > 1) {
      const locale = matched[1];
      if (locale === "en-GB") {
        messages["en-US"] = locales(key).default;
      }
      messages[locale] = locales(key).default;
    }
  });
  return messages;
}
 
export default new VueI18n({
  locale: process.env.VUE_APP_I18N_LOCALE || "en-GB",
  fallbackLocale: process.env.VUE_APP_I18N_FALLBACK_LOCALE || "en-GB",
  messages: loadLocaleMessages(),
});
